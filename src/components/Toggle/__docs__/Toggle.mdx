# Toggle

A switch control component that allows users to turn an option on or off. The Toggle component supports labels, required indicators, info icons, and different sizes.

## Usage

```tsx
import { Toggle } from '@femmecubator/prima-ds';

// Basic usage
<Toggle />

// With label
<Toggle label="Enable notifications" />

// With label and info
<Toggle label="Subscribe to newsletter" info />

// Required toggle
<Toggle label="I agree to terms" required />

// Disabled state
<Toggle label="Disabled option" disabled />

// Different sizes
<Toggle label="Small toggle" size="small" />
<Toggle label="Tiny toggle" size="tiny" />

// Controlled usage
const [enabled, setEnabled] = useState(false);

<Toggle
  label="Feature enabled"
  checked={enabled}
  onChange={(e) => setEnabled(e.target.checked)}
/>

// Form integration with name and value
<form>
  <Toggle
    name="notifications"
    value="email-notifications"
    label="Email notifications"
    checked={emailEnabled}
    onChange={(e) => setEmailEnabled(e.target.checked)}
  />
  <Toggle
    name="preferences"
    value="dark-mode"
    label="Dark mode"
    checked={darkMode}
    onChange={(e) => setDarkMode(e.target.checked)}
  />
</form>

// With explicit ID for accessibility
<Toggle
  id="terms-toggle"
  label="I agree to the terms and conditions"
  required
/>
```
