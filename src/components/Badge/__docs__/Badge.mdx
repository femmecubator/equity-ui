## Usage

```tsx
import { Badge } from '@femmecubator/equity-ui';

// Basic usage
<Badge>Default Badge</Badge>

// With status
<Badge status="success">Completed</Badge>

// With icon
<Badge status="error" icon={true}>Failed</Badge>

// Small size
<Badge status="warning" size="small">Pending</Badge>

// All props
<Badge status="info" size="medium" icon={false}>
  Information
</Badge>
```
