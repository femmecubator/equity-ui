## Usage

```tsx
import { Typography } from '@femmecubator/prima-ds';

// Basic usage
<Typography variant="body" size={2}>Default Body Text</Typography>

// Display text
<Typography variant="display" size={1}>Large Display</Typography>

// Headlines
<Typography variant="headline" size={1}>Main Heading</Typography>

// Labels
<Typography variant="label" size={2}>Form Label</Typography>

// Meta text
<Typography variant="meta" size={1}>META TEXT</Typography>

// Links
<Typography variant="link" size={2}>Click Here</Typography>

// With custom color
<Typography variant="body" size={2} color="#ff0000">
  Custom Color Text
</Typography>

// With custom CSS
<Typography variant="body" size={2} css={{ fontWeight: 'bold', textDecoration: 'underline' }}>
  Bold and Underlined Text
</Typography>
```
