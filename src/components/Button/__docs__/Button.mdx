## Usage

```tsx
import { Button } from '@femmecubator/prima-ds';

// Basic usage
<Button>Click me</Button>

// Different variants
<Button variant="primary">Primary</Button>
<Button variant="secondary">Secondary</Button>
<Button variant="tertiary">Tertiary</Button>

// Different sizes
<Button size="medium">Medium</Button>
<Button size="small">Small</Button>
<Button size="tiny">Tiny</Button>

// Different shapes
<Button shape="pill">Pill Shape</Button>
<Button shape="square">Square Shape</Button>

// Different colors
<Button color="purple">Purple Theme</Button>
<Button color="blue">Blue Theme</Button>

// With icons
<Button iconStart="statistics">Left Icon</Button>
<Button iconEnd="complete">Right Icon</Button>
<Button iconStart="money" iconEnd="globe">Both Icons</Button>

// Disabled state
<Button disabled>Disabled Button</Button>

// Complete example
<Button
  variant="primary"
  color="purple"
  size="medium"
  shape="pill"
  iconStart="statistics"
  onClick={() => handleClick()}
>
  Interactive Button
</Button>
```
