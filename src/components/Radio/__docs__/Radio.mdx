## Usage

```tsx
import { Radio } from '@femmecubator/prima-ds';

// Basic usage
<Radio name="choice" value="option1" />

// With label
<Radio name="choice" value="option1" label="Accept terms" />

// With label and info
<Radio name="choice" value="option1" label="Subscribe to newsletter" info />

// Required radio
<Radio name="choice" value="option1" label="I agree" required />

// Disabled state
<Radio name="choice" value="option1" label="Disabled option" disabled />

// Error state
<Radio name="choice" value="option1" label="Has error" error />

// Group usage
// Vertical group
<Radio.Group
  name="preferences"
  label="Select preference"
  value={selectedValue}
  onChange={setSelectedValue}
  orientation="vertical"
>
  <Radio value="option1" label="Option 1" />
  <Radio value="option2" label="Option 2" />
  <Radio value="option3" label="Option 3" />
</Radio.Group>

// Horizontal group
<Radio.Group
  name="size"
  label="Size"
  value={size}
  onChange={setSize}
  orientation="horizontal"
>
  <Radio value="small" label="Small" />
  <Radio value="medium" label="Medium" />
  <Radio value="large" label="Large" />
</Radio.Group>
```
