## Usage

```tsx
import { Tag } from '@femmecubator/equity-ui';

// Basic usage
<Tag>Category</Tag>

// Different sizes
<Tag size="tiny">Tiny Tag</Tag>
<Tag size="small">Small Tag</Tag>
<Tag size="medium">Medium Tag</Tag>

// Interactive tag
<Tag onClick={() => console.log('clicked')}>
  Clickable Tag
</Tag>

// Active state
<Tag active>Active Tag</Tag>

// Disabled state
<Tag disabled>Disabled Tag</Tag>

// With removable button
<Tag
  withRemovableButton
  onRemove={() => console.log('removed')}
>
  Removable Tag
</Tag>

// Complete example
<Tag
  size="medium"
  active
  withRemovableButton
  onClick={() => handleTagClick()}
  onRemove={() => handleTagRemove()}
>
  Interactive Tag
</Tag>
```
