A versatile checkbox component with support for labels, states, and group functionality using compound component pattern.

## Usage

```tsx
import { Checkbox } from '@femmecubator/equity-ui';

// Basic usage
<Checkbox />

// With label
<Checkbox label="Accept terms" />

// With label and info
<Checkbox label="Subscribe to newsletter" info />

// Required checkbox
<Checkbox label="I agree" required />

// Disabled state
<Checkbox label="Disabled option" disabled />

// Error state
<Checkbox label="Has error" error />

// Indeterminate state
<Checkbox indeterminate />

// Group usage
// Vertical group
<Checkbox.Group
  label="Select options"
  value={selectedValues}
  onChange={setSelectedValues}
  orientation="vertical"
>
  <Checkbox value="option1" label="Option 1" />
  <Checkbox value="option2" label="Option 2" />
  <Checkbox value="option3" label="Option 3" />
</Checkbox.Group>

// Horizontal group
<Checkbox.Group
  label="Categories"
  value={categories}
  onChange={setCategories}
  orientation="horizontal"
>
  <Checkbox value="tech" label="Technology" />
  <Checkbox value="design" label="Design" />
  <Checkbox value="business" label="Business" />
</Checkbox.Group>
```
